 <?php
// admin/insumos.php
session_start();
require_once __DIR__ . '/../backend/conexao.php';

// 🔒 Verifica login
if (!isset($_SESSION['is_admin']) || !$_SESSION['is_admin']) {
    header("Location: login.php");
    exit;
}

// ========================
// PROCESSAR FORMULÁRIOS
// ========================
if ($_SERVER['REQUEST_METHOD'] === 'POST') {
    $nome = trim($_POST['nome'] ?? '');
    $unidade = trim($_POST['unidade'] ?? 'un');
    $preco = floatval($_POST['preco_unit'] ?? 0);
    $estoque = floatval($_POST['estoque'] ?? 0);

    if (isset($_POST['id']) && $_POST['id'] !== '') {
        // Editar
        $id = intval($_POST['id']);
        $stmt = $conn->prepare("UPDATE insumos SET nome=?, unidade=?, preco_unit=?, estoque=? WHERE id=?");
        $stmt->bind_param("ssdsi", $nome, $unidade, $preco, $estoque, $id);
        $stmt->execute();
        $stmt->close();
    } else {
        // Novo
        $stmt = $conn->prepare("INSERT INTO insumos (nome, unidade, preco_unit, estoque) VALUES (?,?,?,?)");
        $stmt->bind_param("ssdd", $nome, $unidade, $preco, $estoque);
        $stmt->execute();
        $stmt->close();
    }
    header("Location: insumos.php");
    exit;
}

// Excluir
if (isset($_GET['del']) && is_numeric($_GET['del'])) {
    $id = intval($_GET['del']);
    $conn->query("DELETE FROM insumos WHERE id=$id");
    header("Location: insumos.php");
    exit;
}

// ========================
// LISTAR INSUMOS
// ========================
$insumos = $conn->query("SELECT * FROM insumos ORDER BY nome ASC")->fetch_all(MYSQLI_ASSOC);

<script>
// Preenche modal para edição
const modal = document.getElementById('modalInsumo');
modal.addEventListener('show.bs.modal', function (event) {
  const button = event.relatedTarget;
  if (!button) return;
  const id = button.getAttribute('data-id') || '';
  document.getElementById('insumo-id').value = id;
  document.getElementById('insumo-nome').value = button.getAttribute('data-nome') || '';
  document.getElementById('insumo-unidade').value = button.getAttribute('data-unidade') || 'un';
  document.getElementById('insumo-preco').value = button.getAttribute('data-preco') || '';
  document.getElementById('insumo-estoque').value = button.getAttribute('data-estoque') || '';
});
</script>
